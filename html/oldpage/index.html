My name is Ril (or Rillo, whatever you like better) and I live in Bavaria, Germany. I am currently 23 years old and I was technically doing tech stuff since I was 7.       This OpenSource Project contains two seperate projects operating in Client/Server mode. On the Client-side it uses a SwiftUI iOS-App running on iOS 15.1. It is planned to use RSA Encryption while also using JWTs for authentication.

function hb(size = 0, useMiB = false) {
    if (typeof size !== "number") throw new Error("Invalid size");
                       
    const unit = [
        ["B", "KiB", "MiB", "GiB", "TiB", "PiB", "EiB", "ZiB", "YiB"],
        ["B", "kB", "MB", "GB", "TB", "PB", "EB", "ZB", "YB"]
    ][useMiB ? 0 : 1];
                       
    const base = useMiB ? 1024 : 1000;
    let c = 0;
    do {
        size /= base;
        c++;
        if (size < base || c >= unit.length - 1) {
            break;
        }
    } while (size > base);
    
    return String(Math.floor(Math.round((size * 100))) / 100 + unit[c]);
}
